#define COMM_LOSS_LIMIT         50      // 100ms 상태정보 연속 미응답 50회(5초) 시 통신 두절 판단 [3]


/* ========================================================================== */
/* 2. 패킷 헤더 및 프로토콜 구조체 (C Structs)                                */
/* ========================================================================== */
// 1바이트 단위 메모리 패킹 적용 (네트워크 통신 필수)
#pragma pack(push, 1)

/**
 * @brief 통신 메시지 공통 헤더 (12 Bytes 고정) [4]
 */
typedef struct {
    uint32_t timestamp;     // 화포통제컴퓨터 Timestamp 반환, 자체 발행 시 0x00000000 [5]
    uint8_t  source_id;     // 송신 장치 ID [5]
    uint8_t  dest_id;       // 수신 장치 ID [5]
    uint8_t  code;          // 메시지 명령어 코드 (ACK 메시지는 0xFF) [5, 6]
    uint8_t  request_ack;   // 0xFF: ACK 미요청, 0x01: ACK 요청함 [5]
                            // (ACK 메시지일 경우 0x10: 정상 ACK, 0x11: NACK) [6]
    uint8_t  priority;      // 우선순위 (작을수록 높음) [5]
    uint8_t  send_count;    // 현재 전송 횟수 (1 ~ 4) [5]
    uint16_t data_length;   // 바로 뒤에 이어질 Data 필드의 크기 (Bytes) [5]
} eth_msg_header_t;

/**
 * @brief 일반 Payload 메시지 구조 (가변 길이, 최대 1024 Bytes) [7]
 */
typedef struct {
    eth_msg_header_t header;
    uint8_t payload;  // 가변 길이 데이터 (Header의 data_length 만큼 유효) [5, 7]
    // 마지막에 2 Bytes Checksum이 위치해야 함 (Checksum = 모든 필드 바이트 합의 최하위 2바이트) [8]
} eth_payload_msg_t;

/**
 * @brief ACK 응답 메시지 구조 (18 Bytes 고정) [4]
 */
typedef struct {
    eth_msg_header_t header; // code = 0xFF, data_length = 4 로 세팅 됨 [6]
    uint16_t code_info;      // 수신하여 ACK/NACK 처리하는 대상 메시지의 코드 [6]
    uint16_t ack_info;       // 0x00: 정상, 0x01: Checksum 오류 등 (사유 코드) [4, 6]
    uint16_t checksum;       // 모든 필드 바이트 합의 최하위 2바이트 [6]
} eth_ack_msg_t;

#pragma pack(pop)


/* ========================================================================== */
/* 3. 통신망 상태 머신 및 전원 시퀀스 구현 가이드 (Pseudo-code)               */
/* ========================================================================== */
/**
 * [ 메시지 특성에 따른 ACK 정책 ] [9]
 * 1. Update (갱신/설정) : ACK 요청 안함 (X), 결과 응답 없음 (X)
 * 2. Request (요청)     : ACK 요청 안함 (X), 결과 응답 있음 (O)
 * 3. Command (명령/구동): ACK 요청 함 (O), 결과 응답 있음 (O)
 * 4. Reflect (주기송신) : ACK 요청 안함 (X), 결과 응답 없음 (X) (예: BIT 결과)
 */

typedef enum {
    STATE_BOOTING = 0,
    STATE_WAIT_BOOT_ACK,  // Boot Done 500ms 반복 전송 상태 [3]
    STATE_JOINED,         // 통신망 가입 완료 (Heartbeat 100ms 전송 상태) [3]
    STATE_COMM_LOSS       // 통신 두절 에러 상태 [3]
} eth_network_state_e;

/* 
 * @brief [백그라운드 Task 구현 가이드] 메인 while(1) 루프의 100ms 타이머 내에서 호출 
 */
void csu_Ethernet_Background_Task(void) {
    static uint32_t loss_counter = 0;
    static uint32_t boot_timer = 0;

    switch(current_net_state) {
        case STATE_BOOTING:
            // 28V 제어전원 인가 -> UDP Bind -> PBIT 완료 후 [10]
            // 제어전원 인가 완료 메시지 전송 [10]
            current_net_state = STATE_WAIT_BOOT_ACK;
            break;
            
        case STATE_WAIT_BOOT_ACK:
            // 500ms 주기 Boot Done 전송 [2]
            if(boot_timer >= 500) {
                Send_Message(BOOT_DONE_MSG);
                boot_timer = 0;
            }
            // ACK를 XINT1 인터럽트에서 수신하면 current_net_state = STATE_JOINED 로 변경됨 [2]
            break;
            
        case STATE_JOINED:
            // 1. 100ms 단위 상태정보(Heartbeat) 통신 개시 [2]
            Send_Message(HEARTBEAT_REQ_MSG);
            
            // 2. ACK/상태정보 미응답 감시 로직 (Timeout) [3]
            if(Message_Not_Received) {
                loss_counter++;
            } else {
                loss_counter = 0; // 수신 시 카운트 초기화
            }
            
            // 3. 통신 두절 판단 [3]
            if(loss_counter >= COMM_LOSS_LIMIT) { // 연속 50회(5초) 미수신 [3]
                Reset_UDP_Socket(); // 통신 리셋 [3]
                current_net_state = STATE_WAIT_BOOT_ACK; // 망 가입 절차로 롤백 [3]
                loss_counter = 0;
            }
            
            // 4. CBIT 송신 (IBIT 중이 아닐 때 지정된 N초 단위로 수행, ACK 미요청) [3, 11]
            if(cbit_flag_enabled && !ibit_running) {
                // CBIT 루틴 실행 및 결과 전송 (Reflect 메시지) [11]
            }
            break;
    }
    
    // (전원 시퀀스 참고) 체계에서 270V 구동전원 인가 메시지가 수신되면, 
    // ACK 응답 후 Heartbeat 상태정보에 "구동전원 인가 상태"를 반영하여 전송해야 함. [10]
}

/*
 * @brief Checksum 연산 유틸리티
 * 체크섬 필드(맨 끝 2바이트)를 제외한 모든 바이트를 더한 후 최하위 2 Bytes를 추출 [6, 8]
 */
uint16_t Calculate_Checksum(uint8_t* packet, uint16_t total_length) {
    uint32_t sum = 0;
    // total_length - 2 위치까지만 합산
    for(uint16_t i = 0; i < total_length - 2; i++) {
        sum += packet[i];
    }
    return (uint16_t)(sum & 0xFFFF);
}
